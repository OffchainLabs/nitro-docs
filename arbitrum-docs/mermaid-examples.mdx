## Example 1

``` mermaid
---
config:
  layout: elk
  look: neo
---

%%{
  init: {
    'theme': 'base',
    'themeVariables': {
      'curve': 'stepBefore',
      'primaryColor': '#BB2528',
      'primaryTextColor': '#fff',
      'primaryBorderColor': '#7C0000',
      'lineColor': '#F8B229',
      'secondaryColor': '#006100',
      'tertiaryColor': '#fff',
      'rankSpacing': '150'
    }
  }
}%%
graph TD
          L3A("Orbit L3")
          L3B("Orbit L3")
          L3C("Orbit L3")
          L3D("Orbit L3")
          L2A("Arbitrum One")
          L2B("Arbitrum Nova")
          L2C("Ethereum L2")
          L2D("Orbit L2")
    L1("Ethereum L1")
    L3A --> L2C
    L3B --> L2C
    L3C --> L2A
    L3D --> L2B
    L2A --> L1
    L2B --> L1
    L2C --> L1
    L2D --> L1 
```
